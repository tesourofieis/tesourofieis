import { getCalendar } from "../../../../lib/getCalendar.ts";

import { yyyyMMDD } from "../../../../lib/src/utils.ts";

import { Card, LinkCard } from "@astrojs/starlight/components";

export const day = yyyyMMDD(new Date());

export const calendar = getCalendar(new Date(day).getFullYear());

<div>
  {Object.entries(calendar).map(([date, celebrations]) => (
    <div
      class={`border rounded p-3 ${date === day ? "bg-red-500 text-white" : ""}`}
    >
      <h2 id={date} />
      {date === day ? (
        <>
          <h5>Hoje</h5>
          <caption>{date}</caption>
        </>
      ) : (
        <caption>{date}</caption>
      )}
      <LinkCard
        href={`/missal/dia#${date}`}
        title={
          celebrations.celebration[0]?.title ||
          celebrations.tempora[0]?.title ||
          celebrations.commemoration[0]?.title ||
          "Feria"
        }
        description={celebrations.commemoration[0]?.title}
      />
      <caption></caption>
    </div>
  ))}
</div>
